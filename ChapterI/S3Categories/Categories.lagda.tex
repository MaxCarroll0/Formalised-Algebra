\subsection*{Agda}

A record of objects, morphisms, identities, composition, and the three
axioms; \AgdaFunction{one} is the terminal category.

\begin{code}
module ChapterI.S3Categories.Categories where

open import Level using (Level; suc; _⊔_; zero)
open import Data.Unit using (⊤; tt)
open import Relation.Binary.PropositionalEquality using (_≡_; refl)

record SmallCategory (o h : Level) : Set (suc (o ⊔ h)) where
  field
    Obj   : Set o
    Hom   : Obj → Obj → Set h
    id    : (X : Obj) → Hom X X
    _∘_   : {X Y Z : Obj} → Hom Y Z → Hom X Y → Hom X Z
    idˡ   : {X Y : Obj} (f : Hom X Y) → (id Y ∘ f) ≡ f
    idʳ   : {X Y : Obj} (f : Hom X Y) → (f ∘ id X) ≡ f
    assoc : {W X Y Z : Obj} (f : Hom W X) (g : Hom X Y) (h : Hom Y Z) →
            ((h ∘ g) ∘ f) ≡ (h ∘ (g ∘ f))

one : SmallCategory zero zero
one = record
  { Obj   = ⊤
  ; Hom   = λ _ _ → ⊤
  ; id    = λ _ → tt
  ; _∘_   = λ _ _ → tt
  ; idˡ   = λ _ → refl
  ; idʳ   = λ _ → refl
  ; assoc = λ _ _ _ → refl
  }
\end{code}
